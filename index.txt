                if(!p.Nombre || p.Nombre.trim() === "") return; // Ignora filas vacías

                const esSkinCare = p.Categoria && p.Categoria.trim() === 'Skin Care';
                const esStarWars = p.Categoria && p.Categoria.trim() === 'Star Wars';

                if(p.Imagen3) miniaturasHtml += `<img class="w-10 h-10 object-cover rounded-lg border border-slate-200 cursor-pointer hover:border-blue-500" src="${p.Imagen3}" onclick="this.closest('.product-card').querySelector('.main-img').src=this.src" onerror="this.style.display='none'">`;

                // Determinar el estilo de la etiqueta (Badge)
                let badgeClass = 'badge-collectibles';
                let badgeIcon = 'fa-star';
                if (esSkinCare) {
                    badgeClass = 'badge-skincare';
                    badgeIcon = 'fa-droplet';
                } else if (esStarWars) {
                    badgeClass = 'badge-starwars';
                    badgeIcon = 'fa-jedi';
                }

                const tarjetaHTML = `
                    <div class="product-card bg-white p-6 rounded-[32px] relative flex flex-col h-full">
                        <span class="absolute top-4 right-4 ${badgeClass} text-[10px] font-bold px-3 py-1 rounded-full uppercase tracking-tighter">

                        </span>
                        
                        <div class="aspect-square rounded-2xl overflow-hidden mb-3 bg-slate-50">
                            <img class="main-img w-full h-full object-cover p-2" src="${p.Imagen1}" alt="${p.Nombre}" onerror="this.src='https://via.placeholder.com/400x400?text=${encodeURIComponent(p.Nombre)}'">
                        </div>
                        <div class="flex gap-2 mb-4 justify-center">
                            ${miniaturasHtml}
                        </div>

                } else if(esStarWars) {
                    starWarsGrid.innerHTML += tarjetaHTML;

                    coleccionablesGrid.innerHTML += tarjetaHTML;

            const searchBar = document.getElementById("search-bar");
            searchBar.classList.toggle("hidden");
            if (!searchBar.classList.contains("hidden")) { document.getElementById("search-input").focus(); }
        }
        function performSearch() {
            const input = document.getElementById("search-input").value;
            if(input.trim() !== "") {
                showToast(`Buscando: "${input}"...`, false);
                setTimeout(() => { toggleSearch(); document.getElementById("search-input").value = ""; }, 1000);
            } else {
                showToast("Por favor, escribe lo que estás buscando.", true);
            }
        }
    </script>
